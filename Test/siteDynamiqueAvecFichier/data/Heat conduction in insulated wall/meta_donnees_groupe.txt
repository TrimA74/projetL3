Dimensionless Equation
\begin{align}
\, \frac{\partial u}{\partial t} 
& = \ \frac{k}{L^{\,2}} \, \frac{\partial^{\,2} u}{\partial x^{\,2}} \,
& t & \ \in \ \big[ \, 0, \, \tau \, \big] \,, \;&  x & \ \in \ \big[ \, 0, \, 1 \, \big] \,, \\[3pt] 
u
& = \ u_{\,\mathrm{L}} 
& t & \ > \ 0\,, \,&  x & = \ 0 \,, \\[3pt]
u
& = \ 1
& t & \ > \ 0\,, \,&  x & = \ 1 \,, \\[3pt]
u & = \ 1 \,,
& t & = \ 0\,, \,&  x & \ \in \ \big[ \, 0, \, 1 \, \big] \,.
\end{align}
