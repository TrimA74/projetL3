Dimensionless Equation
\begin{align}
c \, \frac{\partial u}{\partial t} 
& = \ \frac{\partial^{\,2} u}{\partial x^{\,2}} \,
& t & \ \in \ \big[ \, 0, \, 1 \, \big] \,, \;&  x & \ \in \ \big[ \, 0, \, 1 \, \big] \,, \\[3pt] 
u
& = \ 1 
& t & \ > \ 0\,, \,&  x & = \ 0 \,, \\[3pt]
- \frac{\partial u}{ \partial x} 
& = \ \mathrm{Bi} \cdot \bigl(\, u \ - \ u_{\,\mathrm{\infty}} \, \bigl) 
& t & \ > \ 0\,, \,&  x & = \ 1 \,, \\[3pt]
u & = \ 1 \,,
& t & = \ 0\,, \,&  x & \ \in \ \big[ \, 0, \, 1 \, \big] \,.
\end{align}
